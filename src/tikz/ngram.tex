\newcommand{\ngram}[5]{% outer radius, tips, rot angle, options
    \pgfmathsetmacro{\starangle}{360/#2}
    \pgfmathsetmacro{\innerradius}{#1*sin(90-\starangle)/sin(90+\starangle/2)}
    \tstar{\innerradius}{#1}{#2}{#3}{#4}{#5}
}
